# 🧠 一、抽离核心逻辑（最关键）

## `src/utils/toc-runtime.ts`（新建）

```ts
export function initToc(options: {
  containerSelector?: string;
  contentSelector?: string;
}) {
  const {
    containerSelector = "[data-toc-root]",
    contentSelector = ".markdown-content"
  } = options;

  const container = document.querySelector(containerSelector);
  if (!container) return;

  const getOffset = () => {
    const nav = document.getElementById("navbar-wrapper");
    const h = nav ? nav.getBoundingClientRect().height : 0;
    return Math.max(72, Math.round(h) + 12);
  };

  // ✅ 平滑滚动
  container.querySelectorAll("a[href^='#']").forEach((a) => {
    a.addEventListener("click", (e) => {
      const href = a.getAttribute("href");
      if (!href) return;
      const el = document.querySelector(href);
      if (!el) return;

      e.preventDefault();
      const top =
        window.scrollY + el.getBoundingClientRect().top - getOffset();

      window.scrollTo({ top, behavior: "smooth" });
      history.replaceState(null, "", href);
    });
  });

  // ✅ 高亮逻辑
  const activate = (id: string) => {
    container.querySelectorAll("[data-toc-id]").forEach((n) => {
      n.classList.remove("bg-orange-950/5", "dark:bg-white/10");
      n.removeAttribute("data-active");
    });

    const current = container.querySelector(
      `[data-toc-id="${id}"]`
    ) as HTMLElement;

    if (!current) return;

    current.classList.add("bg-orange-950/5", "dark:bg-white/10");
    current.setAttribute("data-active", "true");

    current.scrollIntoView({
      block: "center",
      behavior: "smooth",
    });

    const parentId = current.getAttribute("data-parent-id");
    if (parentId) {
      const parent = container.querySelector(
        `[data-toc-id="${parentId}"]`
      );
      parent?.classList.add("bg-orange-950/5", "dark:bg-white/10");
      parent?.setAttribute("data-active", "true");
    }
  };

  // ✅ IntersectionObserver
  const headings = document
    .querySelector(contentSelector)
    ?.querySelectorAll("h1, h2, h3, h4");

  if (headings?.length) {
    const io = new IntersectionObserver(
      (entries) => {
        entries.forEach((en) => {
          if (en.isIntersecting) {
            activate(en.target.id);
          }
        });
      },
      { rootMargin: "0px 0px -70% 0px" }
    );

    headings.forEach((h) => io.observe(h));
  }

  // ✅ 折叠
  container.querySelectorAll("[data-toggle-for]").forEach((btn) => {
    btn.addEventListener("click", (e) => {
      e.stopPropagation();

      const id = btn.getAttribute("data-toggle-for");
      const children = container.querySelector(
        `[data-children-for="${id}"]`
      ) as HTMLElement;

      if (!children) return;

      const isHidden = children.style.display === "none";
      children.style.display = isHidden ? "block" : "none";

      const icon = btn.querySelector("svg");
      if (icon) {
        icon.style.transform = isHidden
          ? "rotate(0deg)"
          : "rotate(180deg)";
      }
    });
  });
}
```

import { buildTocTree } from "@utils/toc";

const { headings = [] } = Astro.props;

{tree.map((g) => (

          <a
            href={`#${c.id}`}
            data-toc-id={c.id}
            data-parent-id={g.id}
            class="toc-sub"
          >
            {c.text}
          </a>
        ))}
      </div>
    )}
  </div>
))}
